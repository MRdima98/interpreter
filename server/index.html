Monkey Language

Basic syntax

// Give values names
let age = 1;
let name = "Monkey";
let result = 10 * (20 / 2);
// Arrays
let myArray = [1, 2, 3, 4, 5];
myArray[0] // => 1
// Hashes
let thorsten = {"name": "Thorsten", "age": 28};
thorsten["name"] // => "Thorsten"
// Functions
let add = fn(a, b) { return a + b; };
add(1,2); // => 2
// Builtins
len(myArray) // => 5
puts("Hello") // => Hello
first(myArray) // => 1
last(myArray) // => 5
rest(myArray) // => [2, 3, 4, 5]
push(myArray, 6) // => [1, 2, 3, 4, 5, 6]
          
// Class example
class A {
    let a = 5;

    let method = fn () {
        return 3;
    };
};
let a = new A();
a.a; // => 5
a.method(); // => 3

// Inheritance
class B : A { };
